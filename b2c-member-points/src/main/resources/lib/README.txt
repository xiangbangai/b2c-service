
## 手动安装drools包到本地仓库
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-analytics -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-analytics-4.0.7.jar
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-ant -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-ant-4.0.7.jar
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-compiler -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-compiler-4.0.7.jar
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-core -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-core-4.0.7.jar
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-decisiontables -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-decisiontables-4.0.7.jar
mvn install:install-file -DgroupId=org.drools -DartifactId=drools-jsr94 -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/drools-jsr94-4.0.7.jar
mvn install:install-file -DgroupId=com.huateng -DartifactId=toprules -Dversion=4.0.7 -Dpackaging=jar -Dfile=C:/drools/toprules.jar
